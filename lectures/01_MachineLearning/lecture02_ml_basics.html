Lecture 02: Machine Learning Basics

Overview

This lecture covers the fundamental concepts of machine learning, focusing on function fitting from data, the critical importance of generalization, and the mathematical foundations underlying modern ML approaches.

Core Concepts

Machine Learning as Function Fitting

Machine learning is primarily about finding model parameters (Θ) that minimize a loss function, which measures the discrepancy between predicted and true labels.

import numpy as np
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error
import torch

Data Splitting Strategy

The golden rule: Test data should only be used once for final evaluation!

def demonstrate_data_splitting():
    """Demonstrate proper data splitting for ML"""
    total_samples = 1000
    
    # Typical split ratios
    train_ratio = 0.6
    val_ratio = 0.2
    test_ratio = 0.2
    
    train_size = int(total_samples * train_ratio)
    val_size = int(total_samples * val_ratio)
    test_size = total_samples - train_size - val_size
    
    print("Data Splitting Strategy:")
    print(f"Training: {train_size} samples ({train_ratio*100}%) - Learn model parameters")
    print(f"Validation: {val_size} samples ({val_ratio*100}%) - Tune hyperparameters")
    print(f"Test: {test_size} samples ({test_ratio*100}%) - Final evaluation (use only once!)")
    
    return train_size, val_size, test_size

    return acc_l1, acc_l2

Mathematical Foundation

The linear regression model: Ŷ = Θ₀ + Σ(Θⱼ × xᵢⱼ)

class LinearRegressionFromScratch:
    """Implement linear regression from scratch to understand the math"""
    
    def __init__(self):
        self.weights = None
        self.bias = None
    
    def fit(self, X, y):
        """Analytical solution: Θ* = (X^T * X)^(-1) * X^T * y"""
        # Add bias term (intercept)
        X_with_bias = np.column_stack([np.ones(X.shape[0]), X])
        
        # Analytical solution using normal equation
        XtX = X_with_bias.T @ X_with_bias
        Xty = X_with_bias.T @ y
        
        # Solve: (X^T X) θ = X^T y
        theta = np.linalg.solve(XtX, Xty)
        
        self.bias = theta[0]
        self.weights = theta[1:]
        
        print(f"Learned parameters - Bias: {self.bias:.3f}, Weights: {self.weights}")
    
    def predict(self, X):
        """Make predictions using learned parameters"""
        return self.bias + X @ self.weights
    
    def compute_loss(self, X, y):
        """Compute L2 (quadratic) loss: ||ŷ - y||²"""
        y_pred = self.predict(X)
        loss = np.mean((y_pred - y) ** 2)
        return loss

def compare_loss_functions():
    """Compare L1 vs L2 loss sensitivity to outliers"""
    
    # Create data with outliers
    X_outlier = np.linspace(0, 10, 50).reshape(-1, 1)
    y_clean = 2 * X_outlier.squeeze() + 1
    y_outlier = y_clean.copy()
    
    # Add some outliers
    outlier_indices = [10, 20, 30]
    y_outlier[outlier_indices] += np.array([15, -12, 18])
    
    # L2 Loss (sensitive to outliers)
    lr_l2 = LinearRegression()
    lr_l2.fit(X_outlier, y_outlier)
    
    # L1 Loss (more robust to outliers) - using SGDRegressor with L1 loss
    from sklearn.linear_model import SGDRegressor
    lr_l1 = SGDRegressor(loss='epsilon_insensitive', epsilon=0.1, random_state=42)
    lr_l1.fit(X_outlier, y_outlier)
    
    # Visualize
    plt.figure(figsize=(12, 4))
    
    plt.subplot(1, 2, 1)
    plt.scatter(X_outlier, y_outlier, alpha=0.7)
    plt.scatter(X_outlier[outlier_indices], y_outlier[outlier_indices], color='red', s=100, label='Outliers')
    plt.plot(X_outlier, lr_l2.predict(X_outlier), 'g-', label='L2 Loss (Least Squares)', linewidth=2)
    plt.plot(X_outlier, 2 * X_outlier.squeeze() + 1, 'k--', label='True Function', alpha=0.7)
    plt.title('L2 Loss - Sensitive to Outliers')
    plt.legend()
    
    plt.subplot(1, 2, 2)
    plt.scatter(X_outlier, y_outlier, alpha=0.7)
    plt.scatter(X_outlier[outlier_indices], y_outlier[outlier_indices], color='red', s=100, label='Outliers')
    plt.plot(X_outlier, lr_l1.predict(X_outlier), 'orange', label='L1-like Loss', linewidth=2)
    plt.plot(X_outlier, 2 * X_outlier.squeeze() + 1, 'k--', label='True Function', alpha=0.7)
    plt.title('L1-like Loss - More Robust')
    plt.legend()
    
    plt.tight_layout()
    plt.show()

Challenge 1: Overfitting

def demonstrate_overfitting():
    """Show overfitting with polynomial features"""
    from sklearn.preprocessing import PolynomialFeatures
    from sklearn.pipeline import Pipeline
    
    # Generate simple dataset
    X_simple = np.linspace(0, 1, 20).reshape(-1, 1)
    y_simple = np.sin(2 * np.pi * X_simple).ravel() + np.random.normal(0, 0.1, 20)
    
    # Test different polynomial degrees
    degrees = [1, 3, 15]
    
    plt.figure(figsize=(15, 4))
    
    for i, degree in enumerate(degrees):
        plt.subplot(1, 3, i+1)
        
        # Create polynomial features
        poly_reg = Pipeline([
            ('poly', PolynomialFeatures(degree=degree)),
            ('linear', LinearRegression())
        ])
        
        poly_reg.fit(X_simple, y_simple)
        
        # Plot results
        X_plot = np.linspace(0, 1, 100).reshape(-1, 1)
        y_plot = poly_reg.predict(X_plot)
        
        plt.scatter(X_simple, y_simple, alpha=0.7, label='Training Data')
        plt.plot(X_plot, y_plot, 'r-', linewidth=2, label=f'Degree {degree}')
        plt.plot(X_plot, np.sin(2 * np.pi * X_plot).ravel(), 'g--', alpha=0.7, label='True Function')
        
        # Calculate training error
        train_error = mean_squared_error(y_simple, poly_reg.predict(X_simple))
        plt.title(f'Degree {degree}\nTrain MSE: {train_error:.4f}')
        plt.legend()
    
    plt.tight_layout()
    plt.show()

Challenge 2: Domain Shift

def simulate_domain_shift():
    """Simulate domain shift problem"""
    
    # Training domain: clean, controlled conditions
    X_train_domain = np.random.normal(0, 1, (200, 2))
    y_train_domain = (X_train_domain[:, 0] + X_train_domain[:, 1] > 0).astype(int)
    
    # Test domain: different distribution (shifted mean)
    X_test_domain = np.random.normal(1.5, 1, (100, 2))  # Shifted distribution
    y_test_domain = (X_test_domain[:, 0] + X_test_domain[:, 1] > 0).astype(int)
    
    # Train model on first domain
    from sklearn.linear_model import LogisticRegression
    clf = LogisticRegression()
    clf.fit(X_train_domain, y_train_domain)
    
    # Evaluate on both domains
    train_acc = clf.score(X_train_domain, y_train_domain)
    test_acc = clf.score(X_test_domain, y_test_domain)
    
    print("Domain Shift Example:")
    print(f"Training domain accuracy: {train_acc:.3f}")
    print(f"Test domain accuracy: {test_acc:.3f}")
    print(f"Performance drop due to domain shift: {train_acc - test_acc:.3f}")
    
    return train_acc, test_acc

Summary

This lecture established the fundamental principles of machine learning:

  • Core Paradigm: ML is function fitting from data to minimize prediction errors
  • Generalization: The ultimate goal is performance on unseen data, not training data
  • Data Splitting: Proper train/validation/test splits are crucial for honest evaluation
  • Mathematical Foundation: Understanding loss functions, optimization objectives, and analytical solutions
  • Robustness: Different loss functions have different sensitivities to outliers
  • Common Pitfalls: Overfitting, domain shift, and improper evaluation practices

```python
def simulate_domain_shift():
    """Simulate domain shift problem"""
    
    # Training domain: clean, controlled conditions
    X_train_domain = np.random.normal(0, 1, (200, 2))
    y_train_domain = (X_train_domain[:, 0] + X_train_domain[:, 1] > 0).astype(int)
    
    # Test domain: different distribution (shifted mean)
    X_test_domain = np.random.normal(1.5, 1, (100, 2))  # Shifted distribution
    y_test_domain = (X_test_domain[:, 0] + X_test_domain[:, 1] > 0).astype(int)
    
    # Train model on first domain
    from sklearn.linear_model import LogisticRegression
    clf = LogisticRegression()
    clf.fit(X_train_domain, y_train_domain)
    
    # Evaluate on both domains
    train_acc = clf.score(X_train_domain, y_train_domain)
    test_acc = clf.score(X_test_domain, y_test_domain)
    
    print("Domain Shift Example:")
    print(f"Training domain accuracy: {train_acc:.3f}")
    print(f"Test domain accuracy: {test_acc:.3f}")
    print(f"Performance drop due to domain shift: {train_acc - test_acc:.3f}")
    
    return train_acc, test_acc

simulate_domain_shift()
```
