import torch
import numpy as np
import matplotlib.pyplot as plt

# helper function
def poly_features(x, degree: int):
    # built polynomial features from x (up to degree)
    # x -> [1, x, x^2, ..., x^degree]
    cols = [torch.ones_like(x)]
    for p in range(1, degree + 1):
        cols.append(x ** p)
    return torch.cat(cols, dim=1)

def model_train(model, X, y, epochs=2000, lr=0.05):
    loss_func = torch.nn.MSELoss()
    opt = torch.optim.Adam(model.parameters(), lr=lr)
#    opt = torch.optim.SGD(model.parameters(), lr=lr)
    model.train()
    for _ in range(epochs):
        opt.zero_grad()
        loss = loss_func(model(X), y)
        loss.backward()
        opt.step()
    return loss.item()

def model_predict(model, X):
    model.eval()
    with torch.no_grad():
        return model(X)

def model_test(model, X, y):
    loss_func = torch.nn.MSELoss()
    with torch.no_grad():
        return loss_func(model(X), y).item()

Appendix: More splitting

There is also a powerful (and frequently used) method to conduct the train-test split with sklearn functionality. But this requires changing from torch tensors to numpy arrays (and convert back to tensors).

Code 
from sklearn.model_selection import train_test_split

print(f'x.shape: {x.shape} y.shape {y.shape}')
x_train, x_test, y_train, y_test = train_test_split(
    x.numpy(), y.numpy(), test_size=0.2, random_state=42
)

# Convert back to torch
x_train = torch.tensor(x_train, dtype=torch.float32)
x_test  = torch.tensor(x_test, dtype=torch.float32)
y_train = torch.tensor(y_train, dtype=torch.float32)
y_test  = torch.tensor(y_test, dtype=torch.float32)

print(f'x_train.shape: {x_train.shape} y_train.shape {y_train.shape}')
print(f'x_test.shape: {x_test.shape} y_test.shape {y_test.shape}')

## Appendix: More splitting
There is also a powerful (and frequently used) method to conduct the train-test split with sklearn functionality. But this requires changing from torch tensors to numpy arrays (and convert back to tensors). 
```{python}
#| label: test_train_split_sklearn
from sklearn.model_selection import train_test_split

print(f'x.shape: {x.shape} y.shape {y.shape}')
x_train, x_test, y_train, y_test = train_test_split(
    x.numpy(), y.numpy(), test_size=0.2, random_state=42
)

# Convert back to torch
x_train = torch.tensor(x_train, dtype=torch.float32)
x_test  = torch.tensor(x_test, dtype=torch.float32)
y_train = torch.tensor(y_train, dtype=torch.float32)
y_test  = torch.tensor(y_test, dtype=torch.float32)

print(f'x_train.shape: {x_train.shape} y_train.shape {y_train.shape}')
print(f'x_test.shape: {x_test.shape} y_test.shape {y_test.shape}')
```
