Maximum Likelihood Estimation (MLE)

MLE provides a principled way to derive loss functions by maximizing the probability of observing the training data under our model assumptions.

Mathematical Foundation

import numpy as np
import matplotlib.pyplot as plt
import torch
import torch.nn as nn
import torch.optim as optim
from scipy import stats
import seaborn as sns

def demonstrate_mle_gaussian():
    """Demonstrate MLE for Gaussian distribution leading to L2 loss"""
    
    # Generate synthetic data
    np.random.seed(42)
    true_mean = 5.0
    true_std = 2.0
    n_samples = 100
    
    data = np.random.normal(true_mean, true_std, n_samples)
    
    # MLE for Gaussian parameters
    mle_mean = np.mean(data)
    mle_std = np.std(data, ddof=1)  # Unbiased estimator
    
    print("MLE Demonstration:")
    print(f"True parameters: μ={true_mean}, σ={true_std}")
    print(f"MLE estimates: μ={mle_mean:.3f}, σ={mle_std:.3f}")
    
    # Show how MLE connects to L2 loss
    def negative_log_likelihood(predictions, targets, sigma=1.0):
        """Negative log-likelihood for Gaussian assumption"""
        # -log p(y|x) = -log(1/√(2πσ²)) - (y-ŷ)²/(2σ²)
        # Minimizing NLL ≡ minimizing (y-ŷ)² (L2 loss)
        return 0.5 * np.sum((predictions - targets)**2) / (sigma**2)
    
    # Demonstrate connection
    predictions = np.array([4.8, 5.2, 4.9])
    targets = np.array([5.0, 5.0, 5.0])
    
    nll = negative_log_likelihood(predictions, targets)
    l2_loss = 0.5 * np.sum((predictions - targets)**2)
    
    print(f"\nConnection between MLE and L2 loss:")
    print(f"Negative Log-Likelihood: {nll:.4f}")
    print(f"L2 Loss: {l2_loss:.4f}")
    print("They are equivalent when σ=1!")

From Linear to Logistic Regression

class LogisticRegressionFromScratch:
    """Implement logistic regression to understand sigmoid and cross-entropy"""
    
    def __init__(self, learning_rate=0.01, max_iterations=1000):
        self.lr = learning_rate
        self.max_iter = max_iterations
        self.weights = None
        self.bias = None
        
    def sigmoid(self, z):
        """Sigmoid function: σ(z) = 1/(1 + exp(-z))"""
        # Clip z to prevent overflow
        z = np.clip(z, -250, 250)
        return 1 / (1 + np.exp(-z))
    
    def fit(self, X, y):
        """Train using gradient descent on cross-entropy loss"""
        n_samples, n_features = X.shape
        
        # Initialize parameters
        self.weights = np.zeros(n_features)
        self.bias = 0
        
        # Store loss history
        losses = []
        
        for i in range(self.max_iter):
            # Forward pass
            z = X @ self.weights + self.bias
            y_pred = self.sigmoid(z)
            
            # Binary cross-entropy loss
            # L = -Σ[y*log(ŷ) + (1-y)*log(1-ŷ)]
            epsilon = 1e-15  # Prevent log(0)
            y_pred = np.clip(y_pred, epsilon, 1 - epsilon)
            loss = -np.mean(y * np.log(y_pred) + (1 - y) * np.log(1 - y_pred))
            losses.append(loss)
            
            # Gradients
            dw = (1/n_samples) * X.T @ (y_pred - y)
            db = (1/n_samples) * np.sum(y_pred - y)
            
            # Update parameters
            self.weights -= self.lr * dw
            self.bias -= self.lr * db
            
            if i % 100 == 0:
                print(f"Iteration {i}, Loss: {loss:.4f}")
        
        return losses
    
    def predict_proba(self, X):
        """Return probabilities"""
        z = X @ self.weights + self.bias
        return self.sigmoid(z)
    
    def predict(self, X):
        """Return binary predictions"""
        return (self.predict_proba(X) >= 0.5).astype(int)

# Demonstrate logistic regression
from sklearn.datasets import make_classification

X_class, y_class = make_classification(n_samples=1000, n_features=2, n_redundant=0, 
                                      n_informative=2, n_clusters_per_class=1, random_state=42)

# Train custom logistic regression
custom_lr = LogisticRegressionFromScratch(learning_rate=0.1, max_iterations=1000)
losses = custom_lr.fit(X_class, y_class)

# Compare with sklearn
from sklearn.linear_model import LogisticRegression
sklearn_lr = LogisticRegression()
sklearn_lr.fit(X_class, y_class)

print(f"\nAccuracy comparison:")
print(f"Custom implementation: {np.mean(custom_lr.predict(X_class) == y_class):.3f}")
print(f"Sklearn implementation: {sklearn_lr.score(X_class, y_class):.3f}")
