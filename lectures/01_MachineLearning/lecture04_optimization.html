Lecture 04: Optimization and Backpropagation

Overview

This lecture focuses on the mathematical foundations of neural network training, covering gradient descent optimization, the backpropagation algorithm, and the challenges of optimizing non-convex loss functions in high-dimensional spaces.

Loss Functions and Optimization Objectives

The primary goal of neural network training is to find optimal parameters Θ* that minimize the loss function L(Θ).

import numpy as np
import matplotlib.pyplot as plt
import torch
import torch.nn as nn
import torch.optim as optim
from mpl_toolkits.mplot3d import Axes3D

def visualize_loss_landscapes():
    """Visualize different types of loss landscapes"""
    
    # Create 2D parameter space for visualization
    x = np.linspace(-3, 3, 100)
    y = np.linspace(-3, 3, 100)
    X, Y = np.meshgrid(x, y)
    
    # Convex function (bowl-shaped)
    Z_convex = X**2 + Y**2
    
    # Non-convex function with multiple minima
    Z_nonconvex = np.sin(X) * np.cos(Y) + 0.1 * (X**2 + Y**2)
    
    # Saddle point function
    Z_saddle = X**2 - Y**2
    
    fig = plt.figure(figsize=(15, 5))
    
    # Plot convex surface
    ax1 = fig.add_subplot(131, projection='3d')
    ax1.plot_surface(X, Y, Z_convex, cmap='viridis', alpha=0.7)
    ax1.set_title('Convex Loss (Single Global Minimum)')
    ax1.set_xlabel('Parameter 1')
    ax1.set_ylabel('Parameter 2')
    ax1.set_zlabel('Loss')
    
    # Plot non-convex surface
    ax2 = fig.add_subplot(132, projection='3d')
    ax2.plot_surface(X, Y, Z_nonconvex, cmap='viridis', alpha=0.7)
    ax2.set_title('Non-convex Loss (Multiple Local Minima)')
    ax2.set_xlabel('Parameter 1')
    ax2.set_ylabel('Parameter 2')
    
    # Plot saddle point
    ax3 = fig.add_subplot(133, projection='3d')
    ax3.plot_surface(X, Y, Z_saddle, cmap='viridis', alpha=0.7)
    ax3.set_title('Saddle Point')
    ax3.set_xlabel('Parameter 1')
    ax3.set_ylabel('Parameter 2')
    
    plt.tight_layout()
    plt.show()
    
    print("Loss Landscape Types:")
    print("1. Convex: Single global minimum, gradient descent guaranteed to find it")
    print("2. Non-convex: Multiple local minima, gradient descent finds local minimum")
    print("3. Saddle Point: Minimum in one direction, maximum in another")

Gradient Descent Fundamentals

class GradientDescentOptimizer:
    """Implement gradient descent from scratch"""
    
    def __init__(self, learning_rate=0.01):
        self.lr = learning_rate
        self.loss_history = []
        self.param_history = []
    
    def quadratic_function(self, x, y):
        """Simple quadratic function: f(x,y) = x² + y²"""
        return x**2 + y**2
    
    def gradient_quadratic(self, x, y):
        """Gradient of quadratic function: ∇f = [2x, 2y]"""
        return np.array([2*x, 2*y])
    
    def rosenbrock_function(self, x, y, a=1, b=100):
        """Rosenbrock function: challenging non-convex optimization problem"""
        return (a - x)**2 + b * (y - x**2)**2
    
    def gradient_rosenbrock(self, x, y, a=1, b=100):
        """Gradient of Rosenbrock function"""
        dx = -2*(a - x) - 4*b*x*(y - x**2)
        dy = 2*b*(y - x**2)
        return np.array([dx, dy])
    
    def optimize(self, initial_params, max_iterations=1000, function_type='quadratic'):
        """Run gradient descent optimization"""
        
        params = np.array(initial_params, dtype=float)
        self.param_history = [params.copy()]
        self.loss_history = []
        
        for i in range(max_iterations):
            # Compute loss and gradient
            if function_type == 'quadratic':
                loss = self.quadratic_function(params[0], params[1])
                gradient = self.gradient_quadratic(params[0], params[1])
            elif function_type == 'rosenbrock':
                loss = self.rosenbrock_function(params[0], params[1])
                gradient = self.gradient_rosenbrock(params[0], params[1])
            
            self.loss_history.append(loss)
            
            # Gradient descent update: θ = θ - α∇L(θ)
            params = params - self.lr * gradient
            self.param_history.append(params.copy())
            
            # Print progress
            if i % 100 == 0 or i < 10:
                print(f"Iteration {i:3d}: Loss = {loss:.6f}, "
                      f"Params = [{params[0]:.4f}, {params[1]:.4f}], "
                      f"Gradient norm = {np.linalg.norm(gradient):.6f}")
            
            # Check for convergence
            if np.linalg.norm(gradient) < 1e-6:
                print(f"Converged at iteration {i}")
                break
        
        return params, self.loss_history, self.param_history

def analyze_learning_rates():
    """Analyze the effect of different learning rates"""
    
    learning_rates = [0.001, 0.01, 0.1, 0.5, 1.0, 1.5]
    
    plt.figure(figsize=(15, 10))
    
    for i, lr in enumerate(learning_rates):
        optimizer = GradientDescentOptimizer(learning_rate=lr)
        _, losses, _ = optimizer.optimize(
            initial_params=[2.0, 2.0], 
            max_iterations=50, 
            function_type='quadratic'
        )
        
        plt.subplot(2, 3, i+1)
        plt.plot(losses, 'b-', linewidth=2)
        plt.title(f'Learning Rate = {lr}')
        plt.xlabel('Iteration')
        plt.ylabel('Loss')
        plt.grid(True, alpha=0.3)
        plt.yscale('log')
        
        # Analyze convergence behavior
        if lr <= 0.1:
            behavior = "Stable convergence"
        elif lr <= 0.5:
            behavior = "Fast convergence"
        elif lr <= 1.0:
            behavior = "Oscillatory"
        else:
            behavior = "Divergent"
        
        plt.text(0.5, 0.8, behavior, transform=plt.gca().transAxes, 
                bbox=dict(boxstyle="round,pad=0.3", facecolor="yellow", alpha=0.7))
    
    plt.tight_layout()
    plt.show()
    
    print("Learning Rate Effects:")
    print("• Too small (≤0.01): Slow but stable convergence")
    print("• Optimal (0.1): Fast and stable convergence") 
    print("• Too large (≥1.0): Oscillations or divergence")

class GradientDescentOptimizer:
    """Implement gradient descent from scratch"""
    
    def __init__(self, learning_rate=0.01):
        self.lr = learning_rate
        self.loss_history = []
        self.param_history = []
    
    def quadratic_function(self, x, y):
        """Simple quadratic function: f(x,y) = x² + y²"""
        return x**2 + y**2
    
    def gradient_quadratic(self, x, y):
        """Gradient of quadratic function: ∇f = [2x, 2y]"""
        return np.array([2*x, 2*y])
    
    def rosenbrock_function(self, x, y, a=1, b=100):
        """Rosenbrock function: challenging non-convex optimization problem"""
        return (a - x)**2 + b * (y - x**2)**2
    
    def gradient_rosenbrock(self, x, y, a=1, b=100):
        """Gradient of Rosenbrock function"""
        dx = -2*(a - x) - 4*b*x*(y - x**2)
        dy = 2*b*(y - x**2)
        return np.array([dx, dy])
    
    def optimize(self, initial_params, max_iterations=1000, function_type='quadratic'):
        """Run gradient descent optimization"""
        
        params = np.array(initial_params, dtype=float)
        self.param_history = [params.copy()]
        self.loss_history = []
        
        for i in range(max_iterations):
            # Compute loss and gradient
            if function_type == 'quadratic':
                loss = self.quadratic_function(params[0], params[1])
                gradient = self.gradient_quadratic(params[0], params[1])
            elif function_type == 'rosenbrock':
                loss = self.rosenbrock_function(params[0], params[1])
                gradient = self.gradient_rosenbrock(params[0], params[1])
            
            self.loss_history.append(loss)
            
            # Gradient descent update: θ = θ - α∇L(θ)
            params = params - self.lr * gradient
            self.param_history.append(params.copy())
            
            # Print progress
            if i % 100 == 0 or i < 10:
                print(f"Iteration {i:3d}: Loss = {loss:.6f}, "
                      f"Params = [{params[0]:.4f}, {params[1]:.4f}], "
                      f"Gradient norm = {np.linalg.norm(gradient):.6f}")
            
            # Check for convergence
            if np.linalg.norm(gradient) < 1e-6:
                print(f"Converged at iteration {i}")
                break
        
        return params, self.loss_history, self.param_history

    def __init__(self, input_size, hidden_size, output_size):
        # Initialize weights randomly
        self.W1 = np.random.randn(input_size, hidden_size) * 0.1
        self.b1 = np.zeros((1, hidden_size))
        self.W2 = np.random.randn(hidden_size, output_size) * 0.1
        self.b2 = np.zeros((1, output_size))
        
        # Store intermediate values for backprop
        self.z1 = None
        self.a1 = None
        self.z2 = None
        self.a2 = None
        
    def sigmoid(self, x):
        """Sigmoid activation function"""
        return 1 / (1 + np.exp(-np.clip(x, -250, 250)))
