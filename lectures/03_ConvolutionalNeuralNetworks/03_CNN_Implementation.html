Downsampling and Splitting

Downsampling is done for faster learning, but at the expense of accuracy

Code 
from sklearn.model_selection import train_test_split
from torch.utils.data import Subset

FRACT = 0.1   # fraction to sample
TRAIN_VAL=0.8 # train-validation split

# get all indices and labels
train_indices = np.arange(len(train_dataset)) 
train_labels  = np.array(train_dataset.targets)
test_indices  = np.arange(len(test_dataset))  
test_labels   = np.array(test_dataset.targets)

# subsample fraction FRACT from train and test data 
# here just for speed, usually we want more data
train_pool_idx, _ = train_test_split(
    train_indices, train_size=FRACT, stratify=train_labels,
    random_state=SEED
)
test_idx, _ = train_test_split(
    test_indices, train_size=FRACT, stratify=test_labels, 
    random_state=SEED
)

# stratified split: train_pool -> train + validation
y_pool = train_labels[train_pool_idx]
train_idx, val_idx = train_test_split(
    train_pool_idx, train_size=TRAIN_VAL, stratify=y_pool,
    random_state=SEED
)

# define subsets
train_ds = Subset(train_dataset, train_idx)
val_ds   = Subset(train_dataset, val_idx)
test_ds  = Subset(test_dataset,  test_idx)

print(f"train: {train_idx.shape}, val: {val_idx.shape}, test: {test_idx.shape}")

Model Definition (CNN)

Define Model Class

It will be useful to define the model class in a separate file that can be used reproducibly across contexts and notebooks.

Here I define a very simple CNN with a single convolutional layer.

Content of models/cnn.py:

import torch
import torch.nn as nn
import torch.nn.functional as F

class CNN_Model(nn.Module):
    def __init__(self, num_classes):
        super().__init__()
        self.conv1 = nn.Conv2d(3, 32, kernel_size=3, padding=1)
        self.pool1 = nn.MaxPool2d(2, 2)
        self.flatten = nn.Flatten()
        self.fc1 = nn.Linear(32 * 16 * 16, 128)
        self.fc2 = nn.Linear(128, num_classes)

    def forward(self, x):
        x = F.relu(self.conv1(x))
        x = self.pool1(x)
        x = self.flatten(x)
        x = F.relu(self.fc1(x))
        x = self.fc2(x)
        return x

Model Training

Training will take some time - especially without dedicated hardware (e.g. GPU).

In the following the number of epochs is kept low

Code 
NUM_EPOCHS = 15 # keep small for faster execution
history = {'loss': [], 'val_loss': []}
for epoch in range(NUM_EPOCHS):
    model.train()
    train_loss = 0
    # loop over batches
    for xb, yb in train_loader:
        # send batches to device
        xb, yb = xb.to(device), yb.to(device)
        optimizer.zero_grad()
        z_pred = model(xb)
        loss = loss_function(z_pred, yb)
        loss.backward()
        optimizer.step()
        # loss is the _average_ loss per batch
        train_loss += loss.item()

    # average loss over all batches
    train_loss /= len(train_loader)
    history['loss'].append(train_loss)

    # Validation
    model.eval()
    val_loss = 0
    with torch.no_grad():
        for xb, yb in val_loader:
            xb, yb = xb.to(device), yb.to(device)
            z_pred = model(xb)
            loss = loss_function(z_pred, yb)
            val_loss += loss.item()
    val_loss = val_loss / len(val_loader)
    history['val_loss'].append(val_loss)
    print(f"Epoch {epoch+1}: train_loss={train_loss:.4f}, val_loss={val_loss:.4f}")
