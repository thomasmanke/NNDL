def plot_fit_history(hist, name='loss', test_score=None):
    if name not in hist:
        print(f"{name} not found in history")
        return
    plt.plot(hist[name], label='train')
    val_name = 'val_' + name
    if val_name in hist:
        plt.plot(hist[val_name], label='valid')
    if test_score is not None:
        plt.axhline(test_score, color='green', linestyle='-.', label='test')
    plt.xlabel('Epoch')
    plt.ylabel(name)
    plt.legend()
    plt.show()

def dataset2images(dataset, classes, SEED=42, n=16):
    """Show n random images from a dataset in a 4x4 grid."""
    np.random.seed(SEED)
    indices = np.random.choice(len(dataset), n, replace=False)

    plt.figure(figsize=(8, 8))
    for i, idx in enumerate(indices):
        img, label = dataset[idx]

        if torch.is_tensor(img):
            # PyTorch tensor (C,H,W) -> numpy (H,W,C)
            img_np = img.permute(1, 2, 0).cpu().numpy()

        elif isinstance(img, Image.Image):
            # PIL Image
            img_np = np.array(img)

        elif isinstance(img, np.ndarray):
            # numpy array already OK
            img_np = img

        else:
            raise TypeError(
                f"Unsupported image type: {type(img)}. "
                "Expected torch.Tensor, PIL.Image, or numpy.ndarray."
            )

        # Plot
        ax = plt.subplot(4, 4, i + 1)
        ax.imshow(img_np)
        ax.set_title(f"{label}: {classes[label]}")
        ax.axis("off")

    plt.tight_layout()
    plt.show()

Datasets & Benchmarks

  • ImageNet Large Scale Visual Recognition Challenge
  • ILSVRC (2010 - 2017)
  • Human Accuracy (Top-5): 95% (after training!)

AlexNet (2012)

Innovations
  • GPU/CUDA
  • ReLU
  • data augmentation + dropout
  • parameters: 60M
  • accuracy (top-5): 79%

VGGNet (2014)

Innovations
  • simplicity, standardization and modularity
  • performance degradation for deeper layers
  • params: 138M (VGG-16) - 140M (VGG-19)
  • accuracy: 90%

ResNet50 (He et al. 2015)

Innovations
  • Residual Links (“skip connections”) to stabilize learning
  • gradient control for very deep networks
  • 23M params
  • accuracy: 93%

GoogLeNet (Inception V1-V4, 2014 - 2017)

Innovations
  • parallel convolutions (1x1, 3x3, 5x5) + concatenation
  • 1x1 convolutions to reduce channel depth
  • params: 27M
  • accuracy: 93%

Reference: Model Parameters, Accuracy & Links

Transfer Learning

Small Data & Big Model
  • fine-tune only the last layers for domain-specfic tasks
  • basic filters and parameters have been well trained on massive data
  • succesful edge detection, fundamental shapes, etc.
  • specific tasks we only need to update the latter (“conceptual”) layers.

def denormalize(imgs, mean=IMAGENET_MEAN, std=IMAGENET_STD):
    """
    imgs: _normalized_ tensor of shape (B, C, H, W) or (C, H, W)
    mean, std: either scalars or sequences of length C
    Assumes imgs were normalized as (x - mean) / std.
    Purpose: visualize normalized imgs
    """
    # unsqueeze images if not already batch
    is_batched = (imgs.dim() == 4)
    if not is_batched:
        imgs = imgs.unsqueeze(0)   # (C,H,W) -> (1,C,H,W)

    B, C, H, W = imgs.shape

    # Turn mean/std into tensors on the correct device
    if isinstance(mean, (int, float)):
        mean = [mean] * C
    if isinstance(std, (int, float)):
        std = [std] * C

    mean = torch.tensor(mean, device=imgs.device, dtype=imgs.dtype).view(1, C, 1, 1)
    std  = torch.tensor(std,  device=imgs.device, dtype=imgs.dtype).view(1, C, 1, 1)

    imgs_denorm = imgs * std + mean
    imgs_denorm = imgs_denorm.clamp(0, 1)

    if not is_batched:
        imgs_denorm = imgs_denorm.squeeze(0)

    return imgs_denorm

Transfer Model

… fix all parameters apart from new last layer (the “head”)

Note
  • new number of classes for specific problem
  • can replace or unfreeze any layer (requires_grad=True)
Code 
from torchvision.models import resnet50, ResNet50_Weights
from torchinfo import summary

def build_resnet50_transfer(num_classes=1, freeze_backbone=True):
    # Load pretrained weights
    weights = ResNet50_Weights.IMAGENET1K_V1
    model = resnet50(weights=weights)

    # freeze all parameters
    # do not calculate gradients, do not update
    if freeze_backbone:
        for param in model.parameters():
            param.requires_grad = False

    # Replace final fully connected layer
    # the new layer will have unfrozen (trainable) parameters
    in_features = model.fc.in_features
    model.fc = nn.Linear(in_features, num_classes)

    return model

Training

Code 
NUM_EPOCHS = 10 # keep small for faster execution
history = {'loss': [], 'val_loss': []}
# loop over epochs
for epoch in range(NUM_EPOCHS):
    model.train()
    total_loss = 0
    # loop over batches
    for xb, yb in train_loader:
        # send batches to device
        xb, yb = xb.to(device), yb.to(device)
        optimizer.zero_grad()
        z_pred = model(xb)
        loss = loss_function(z_pred, yb)
        loss.backward()
        optimizer.step()
        total_loss += loss.item()

    # record average loss over all batches
    train_loss = total_loss / len(train_loader.dataset)
    history['loss'].append(train_loss)

    # Validation
    model.eval()
    val_loss = 0
    with torch.no_grad():
        for xb, yb in val_loader:
            xb, yb = xb.to(device), yb.to(device)
            z_pred = model(xb)
            loss = loss_function(z_pred, yb)
            val_loss += loss.item()
    val_loss = val_loss / len(val_loader.dataset)
    history['val_loss'].append(val_loss)
    print(f"Epoch {epoch+1}: train_loss={train_loss:.4f}, val_loss={val_loss:.4f}")

```{python}
def plot_fit_history(hist, name='loss', test_score=None):
    if name not in hist:
        print(f"{name} not found in history")
        return
    plt.plot(hist[name], label='train')
    val_name = 'val_' + name
    if val_name in hist:
        plt.plot(hist[val_name], label='valid')
    if test_score is not None:
        plt.axhline(test_score, color='green', linestyle='-.', label='test')
    plt.xlabel('Epoch')
    plt.ylabel(name)
    plt.legend()
    plt.show()

def dataset2images(dataset, classes, SEED=42, n=16):
    """Show n random images from a dataset in a 4x4 grid."""
    np.random.seed(SEED)
    indices = np.random.choice(len(dataset), n, replace=False)

    plt.figure(figsize=(8, 8))
    for i, idx in enumerate(indices):
        img, label = dataset[idx]

        if torch.is_tensor(img):
            # PyTorch tensor (C,H,W) -> numpy (H,W,C)
            img_np = img.permute(1, 2, 0).cpu().numpy()

        elif isinstance(img, Image.Image):
            # PIL Image
            img_np = np.array(img)

        elif isinstance(img, np.ndarray):
            # numpy array already OK
            img_np = img

        else:
            raise TypeError(
                f"Unsupported image type: {type(img)}. "
                "Expected torch.Tensor, PIL.Image, or numpy.ndarray."
            )

        # Plot
        ax = plt.subplot(4, 4, i + 1)
        ax.imshow(img_np)
        ax.set_title(f"{label}: {classes[label]}")
        ax.axis("off")

    plt.tight_layout()
    plt.show()
```

```{python}
#| label: define_transform
IMAGENET_MEAN = (0.485, 0.456, 0.406)
IMAGENET_STD  = (0.229, 0.224, 0.225)
IMAGENET_SIZE = 224

tfm = transforms.Compose([
    # transform to ImageNet Parameters to be used with ResNet
    transforms.Resize(256),
    transforms.CenterCrop(IMAGENET_SIZE),
    transforms.ToTensor(),
    transforms.Normalize(IMAGENET_MEAN, IMAGENET_STD),
])

tfm_aug = transforms.Compose([
    # transform and augment
    transforms.Resize(256),
    transforms.RandomRotation(degrees=20), # augmentation
    transforms.CenterCrop(IMAGENET_SIZE),
    transforms.ToTensor(),
    transforms.Normalize(IMAGENET_MEAN, IMAGENET_STD),
])



def denormalize(imgs, mean=IMAGENET_MEAN, std=IMAGENET_STD):
    """
    imgs: _normalized_ tensor of shape (B, C, H, W) or (C, H, W)
    mean, std: either scalars or sequences of length C
    Assumes imgs were normalized as (x - mean) / std.
    Purpose: visualize normalized imgs
    """
    # unsqueeze images if not already batch
    is_batched = (imgs.dim() == 4)
    if not is_batched:
        imgs = imgs.unsqueeze(0)   # (C,H,W) -> (1,C,H,W)

    B, C, H, W = imgs.shape

    # Turn mean/std into tensors on the correct device
    if isinstance(mean, (int, float)):
        mean = [mean] * C
    if isinstance(std, (int, float)):
        std = [std] * C

    mean = torch.tensor(mean, device=imgs.device, dtype=imgs.dtype).view(1, C, 1, 1)
    std  = torch.tensor(std,  device=imgs.device, dtype=imgs.dtype).view(1, C, 1, 1)

    imgs_denorm = imgs * std + mean
    imgs_denorm = imgs_denorm.clamp(0, 1)

    if not is_batched:
        imgs_denorm = imgs_denorm.squeeze(0)

    return imgs_denorm
```

### Transfer Model

... fix all parameters apart from new last layer (the "head")

::: {.callout-note}
- new number of classes for specific problem
- can replace or unfreeze any layer (`requires_grad=True`)
:::

```{python}
#| label: resnet50_transfer
from torchvision.models import resnet50, ResNet50_Weights
from torchinfo import summary

def build_resnet50_transfer(num_classes=1, freeze_backbone=True):
    # Load pretrained weights
    weights = ResNet50_Weights.IMAGENET1K_V1
    model = resnet50(weights=weights)

    # freeze all parameters
    # do not calculate gradients, do not update
    if freeze_backbone:
        for param in model.parameters():
            param.requires_grad = False

    # Replace final fully connected layer
    # the new layer will have unfrozen (trainable) parameters
    in_features = model.fc.in_features
    model.fc = nn.Linear(in_features, num_classes)

    return model

# instantiate transfered model
```

```{python}
NUM_EPOCHS = 10 # keep small for faster execution
history = {'loss': [], 'val_loss': []}
# loop over epochs
for epoch in range(NUM_EPOCHS):
    model.train()
    total_loss = 0
    # loop over batches
    for xb, yb in train_loader:
        # send batches to device
        xb, yb = xb.to(device), yb.to(device)
        optimizer.zero_grad()
        z_pred = model(xb)
        loss = loss_function(z_pred, yb)
        loss.backward()
        optimizer.step()
        total_loss += loss.item()

    # record average loss over all batches
    train_loss = total_loss / len(train_loader.dataset)
    history['loss'].append(train_loss)

    # Validation
    model.eval()
    val_loss = 0
    with torch.no_grad():
        for xb, yb in val_loader:
            xb, yb = xb.to(device), yb.to(device)
            z_pred = model(xb)
            loss = loss_function(z_pred, yb)
            val_loss += loss.item()
    val_loss = val_loss / len(val_loader.dataset)
    history['val_loss'].append(val_loss)
    print(f"Epoch {epoch+1}: train_loss={train_loss:.4f}, val_loss={val_loss:.4f}")
```
