Building Larger Models

We can define our own models from building blocks. Ultimately this will help to abstract from individual layers, their initialization, normalization, dropouts etc.

Code 
import torch.nn as nn

class MyModel(nn.Module):
    def __init__(self, n_in, n_out):
        # inherit stuff from nn.Module
        super().__init__()
        # define layers: just some random examples
        self.linear = nn.Linear(n_in, n_out)
        self.norm = nn.Dropout(p=0.3)

    # conncect layers defined above
    def forward(self, x):
        x = self.linear(x)
        x = self.norm(x)
        x = nn.functional.sigmoid(x)
        return x

model = MyModel(10, 1)
print(model)
MyModel(
  (linear): Linear(in_features=10, out_features=1, bias=True)
  (norm): Dropout(p=0.3, inplace=False)
)

Optimizer and Loss Function

Code 
import torch.optim as optim

learning_rate = 0.01 # hyperparameter

optimizer = optim.Adam(model.parameters(), lr = learning_rate)

loss_function = nn.MSELoss()

model.train() vs. model.eval()

two different modes for training and evaluation

Feature model.train() (Training Mode) model.eval() (Evaluation Mode)
Dropout Randomly disables neurons (adds noise) All neurons active (no dropout)
BatchNorm Uses current batch stats (mean/var) Uses running (saved) stats
Gradient Tracking Yes Use with torch.no_grad() to disable
When to use During training loop During validation/testing/inference

Typical use:

Code 
# train data
n_samples, m_features = 100, 10
x = torch.rand((n_samples, m_features))
y = torch.bernoulli(x[:,0], 0.5).reshape(-1,1)

# Training
n_epochs = 10
model.train()
for epoch in range(n_epochs):

    output = model(x)           # forward path x -> output

    loss = loss_function(output, y) # loss calculation

    optimizer.zero_grad() # reset gradients (do not accumulate)
    loss.backward()       # backward path L -> x (gradient calc)
    optimizer.step()      # update parm = parm - learning_rate * grad

    # print loss
    if(epoch % 20 == 0):
        print('epoch {}, loss {}'.format(epoch, loss.data))

# Evaluation (Validation or Test or Prediction)
x_val = torch.rand((n_samples, m_features))
y_val = torch.bernoulli(x_val[:,0], 0.5).reshape(-1,1)

model.eval()
# disable grad calculation for speed
with torch.no_grad():
    val_output = model(x_val)
    val_loss = loss_function(val_output, y_val)
epoch 0, loss 0.26598238945007324

Saving models

You may want to save (and share) your final model. But for long-running jobs it may also be good to save occasional checkpoints during training

Code 
import os
save_dir = "checkpoints"
os.makedirs(save_dir, exist_ok=True)

# 1. only saves state_dict, need to know model class
torch.save(model.state_dict(), os.path.join(save_dir, "model_dict.pt"))
model.load_state_dict(torch.load("checkpoints/model_dict.pt"))

# 2. save full model. Warning: may not port across pytorch versions!!! 
torch.save(model, os.path.join(save_dir, "model_full.pth"))
model = torch.load("checkpoints/model_full.pth", weights_only=False)

Saving histories

Code 
# 1. npz format (most general and robust)
np.savez(os.path.join(save_dir, "loss_history.npz"),
         train=train_losses, val=val_losses)

data = np.load("checkpoints/loss_history.npz")
train_losses = data["train"]
val_losses = data["val"]

# 2. pickle format (more flexible less portable)
with open(os.path.join(save_dir, "loss_history.pkl"), "wb") as f:
    pickle.dump({"train": train_losses, "val": val_losses}, f)

with open("checkpoints/loss_history.pkl", "rb") as f:
    losses = pickle.load(f)
train_losses = losses["train"]
val_losses = losses["val"]

### Building Larger Models

We can define our own models from building blocks.
Ultimately this will help to abstract from individual layers, their 
initialization, normalization, dropouts etc.

```{python}
import torch.nn as nn

class MyModel(nn.Module):
    def __init__(self, n_in, n_out):
        # inherit stuff from nn.Module
        super().__init__()
        # define layers: just some random examples
        self.linear = nn.Linear(n_in, n_out)
        self.norm = nn.Dropout(p=0.3)

    # conncect layers defined above
    def forward(self, x):
        x = self.linear(x)
        x = self.norm(x)
        x = nn.functional.sigmoid(x)
        return x

model = MyModel(10, 1)
print(model)
```

## Optimizer and Loss Function
```{python}
import torch.optim as optim

learning_rate = 0.01 # hyperparameter

optimizer = optim.Adam(model.parameters(), lr = learning_rate)

loss_function = nn.MSELoss()
```


## model.train() vs. model.eval()

two different modes for training and evaluation

| Feature               | `model.train()` (Training Mode)           | `model.eval()` (Evaluation Mode)         |
| --------------------- | ----------------------------------------- | ---------------------------------------- |
| **Dropout**           | Randomly disables neurons (adds noise) | All neurons active (no dropout)        |
| **BatchNorm**         | Uses current batch stats (mean/var)     | Uses running (saved) stats            |
| **Gradient Tracking** | Yes                                     | Use `with torch.no_grad()` to disable |
| **When to use**       | During training loop                    | During validation/testing/inference    |

Typical use:
```{python}
#| label: model_train_eval

# train data
n_samples, m_features = 100, 10
x = torch.rand((n_samples, m_features))
y = torch.bernoulli(x[:,0], 0.5).reshape(-1,1)

# Training
n_epochs = 10
model.train()
for epoch in range(n_epochs):

    output = model(x)           # forward path x -> output

    loss = loss_function(output, y) # loss calculation

    optimizer.zero_grad() # reset gradients (do not accumulate)
    loss.backward()       # backward path L -> x (gradient calc)
    optimizer.step()      # update parm = parm - learning_rate * grad

    # print loss
    if(epoch % 20 == 0):
        print('epoch {}, loss {}'.format(epoch, loss.data))

# Evaluation (Validation or Test or Prediction)
x_val = torch.rand((n_samples, m_features))
y_val = torch.bernoulli(x_val[:,0], 0.5).reshape(-1,1)

model.eval()
# disable grad calculation for speed
with torch.no_grad():
    val_output = model(x_val)
    val_loss = loss_function(val_output, y_val)

```

## Saving models

You may want to save (and share) your final model.
But for long-running jobs it may also be good to save
occasional checkpoints during training
```{python}
#| label: saving_models
#| eval: false
import os
save_dir = "checkpoints"
os.makedirs(save_dir, exist_ok=True)

# 1. only saves state_dict, need to know model class
torch.save(model.state_dict(), os.path.join(save_dir, "model_dict.pt"))
model.load_state_dict(torch.load("checkpoints/model_dict.pt"))

# 2. save full model. Warning: may not port across pytorch versions!!! 
torch.save(model, os.path.join(save_dir, "model_full.pth"))
model = torch.load("checkpoints/model_full.pth", weights_only=False)
```
