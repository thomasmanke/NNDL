```{python}
#| label: autoencoder_linear
class AutoencoderLinear(nn.Module):
    def __init__(self, nz = 3):
        super().__init__()

        self.encoder = nn.Sequential(
            nn.Linear(28 * 28, 128), # (N, 784) -> (N, 128)
            nn.ReLU(),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Linear(64, 16),
            nn.ReLU(),
            nn.Linear(16, nz) # -> (N, nz)
        )
        
        self.decoder = nn.Sequential(
            nn.Linear(nz, 16), # (N, 16)
            nn.ReLU(),
            nn.Linear(16, 64),
            nn.ReLU(),
            nn.Linear(64, 128),
            nn.ReLU(),
            nn.Linear(128, 28 * 28),
            nn.Sigmoid()
        )

    def forward(self, x):
        encoded = self.encoder(x)
        decoded = self.decoder(encoded)
        return decoded, encoded

```


```{python}
class AutoencoderConv(nn.Module):
    def __init__(self):
        super().__init__()        
        # N, 1, 28, 28
        self.encoder = nn.Sequential(
            nn.Conv2d(1, 16, 3, stride=2, padding=1), # -> N, 16, 14, 14
            nn.ReLU(),
            nn.Conv2d(16, 32, 3, stride=2, padding=1), # -> N, 32, 7, 7
            nn.ReLU(),
            nn.Conv2d(32, 64, 7) # -> N, 64, 1, 1
        )
        
        # N , 64, 1, 1
        self.decoder = nn.Sequential(
            nn.ConvTranspose2d(64, 32, 7), # -> N, 32, 7, 7
            nn.ReLU(),
            nn.ConvTranspose2d(32, 16, 3, stride=2, padding=1, output_padding=1), # N, 16, 14, 14 (N,16,13,13 without output_padding)
            nn.ReLU(),
            nn.ConvTranspose2d(16, 1, 3, stride=2, padding=1, output_padding=1), # N, 1, 28, 28  (N,1,27,27)
            nn.Sigmoid()
        )

    def forward(self, x):
        encoded = self.encoder(x)
        decoded = self.decoder(encoded)
        return decoded, encoded
    
 
# Note: nn.MaxPool2d -> use nn.MaxUnpool2d, or use different kernelsize, stride etc to compensate...
# Input [-1, +1] -> use nn.Tanh
```

```{python}
device = torch.device("mps" if torch.backends.mps.is_available() else "cpu")
model = AutoencoderLinear(nz=2).to(device)
criterion = nn.MSELoss()
optimizer = torch.optim.Adam(model.parameters(), lr=1e-3, weight_decay=1e-5)
```

```{python}
#| label: train_loop
num_epochs = 10
drop_percent = 0.1
outputs = []
for epoch in range(num_epochs):

    for (img, label) in train_loader:
        img = img.reshape(-1, 28*28).to(device) # -> use for Autoencoder_Linear

        mask = (torch.rand_like(img) > drop_percent).float().to(device)
        img_dropped = img * mask

        recon, z = model(img_dropped)
        loss = criterion(recon, img)
        
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

    print(f'Epoch:{epoch}, Loss:{loss.item():.4f}')
    # append output for this epoch (only last batch)
    outputs.append((epoch, 
      img.detach().cpu(), 
      recon.detach().cpu(), 
      z.detach().cpu(), 
      img_dropped.detach().cpu()))
```

```{python}
#| label: plot_perturbations

def plot_latent_perturbations(z0, model, axis=0, num_steps=8, eps_start=-1.0, eps_stop=1.0, device='cpu'):
    """
    Plot original and perturbed decodings from latent space.
    """
    model.eval()
    epsilons = np.linspace(eps_start, eps_stop, num_steps)

    fig, axes = plt.subplots(1, num_steps + 1, figsize=(2 * (num_steps + 1), 2))

    # Original image
    with torch.no_grad():
        img = model.decoder(z0).reshape(-1,28,28).squeeze().cpu().numpy()
        axes[0].imshow(img, cmap='gray')
        axes[0].set_title("original")
        axes[0].axis("off")

    # Perturbed images
    for i,eps in enumerate(epsilons):
        z_eps = z0.clone().detach()
        z_eps[axis] = z0[axis] + eps
        print(i, eps, z0, z_eps)
        with torch.no_grad():
            img = model.decoder(z_eps).reshape(-1,28,28).squeeze().cpu().numpy()
        axes[i + 1].imshow(img, cmap='gray')
        axes[i + 1].set_title(f"ε = {eps:.2f}")
        axes[i + 1].axis("off")

    plt.tight_layout()
    plt.show()
```
